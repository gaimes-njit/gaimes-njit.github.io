```
ab_drop('b13', 1)
ab_drop('b31', 1)

ab_drop('b31', 1)
ab_drop('b31', 1)
ab_drop('b11', 2)
ab_drop('b11', 3)
ab_drop('b11', 4)
ab_drop('b11', 5)

```
