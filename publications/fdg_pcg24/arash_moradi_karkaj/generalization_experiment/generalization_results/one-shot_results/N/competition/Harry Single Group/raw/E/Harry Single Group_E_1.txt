# Start by initializing an empty field
field = [[' ' for _ in range(20)] for _ in range(16)]

def ab_drop(x, y):
    # Determine the dimensions of the brick based on its type
    if x == 'b31':
        width = 3
        height = 1
    elif x == 'b13':
        width = 1
        height = 3
    elif x == 'b11':
        width = 1
        height = 1
        
    # Check if the brick will exceed the boundaries of the field
    if y + width > 20:
        print("Error: Brick exceeds field boundaries.")
        return
    
    # Start dropping the brick from the top of the field
    for i in range(height):
        # Check if any cells in the current position are already occupied
        for j in range(y, y + width):
            if field[i][j] != ' ':
                # Stop the brick above the occupied cell
                if i == 0:
                    print("Error: Brick cannot land on occupied cells.")
                    return
                else:
                    # Adjust the position to stop the brick above the occupied cell
                    height = i
                    break
        else:
            # If no occupied cells are detected in the current position, continue dropping
            continue
        break
    
    # Place the brick in the field
    for i in range(height):
        for j in range(y, y + width):
            field[i][j] = 'X'

# Generate the required E shape
ab_drop('b31', 2)
ab_drop('b11', 2)
ab_drop('b11', 2)
ab_drop('b31', 7)
ab_drop('b31', 12)
ab_drop('b11', 12)
ab_drop('b11', 12)
ab_drop('b31', 17)

# Print the final field
for row in field:
    print(' '.join(row))
