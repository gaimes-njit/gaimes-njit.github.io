# Initialize the map
field = [[0 for _ in range(20)] for _ in range(16)]

            if field[y][x] == 1:

    return True

# Function to place a brick on the field
def place_brick(x_position, y_position, width, height):
    for x in range(x_position, x_position+width):
        for y in range(y_position, y_position+height):
            field[y][x] = 1

# Function to drop a brick and check stability
