# Map initialization
map = [[0 for _ in range(20)] for _ in range(16)]

# Function to check stability
def check_stability():
    stable = True
    for y in range(1, 16):
        pivot = None
        for x in range(20):
            if map[y][x] != 0:
                if pivot == None:
                    pivot = x
                elif x < pivot:
                    stable = False
                    break
                elif x > pivot:
                    stable = False
                    break
    return stable

# Function to place bricks
def place_bricks():
    # Place vertical brick 'b13' at x_position = 5
    ab_drop('b13', 5)
    
    # Place horizontal brick 'b31' at x_position = 6
    ab_drop('b31', 6)
    
    # Check stability and fix if needed
    if not check_stability():
        # Move 'b13' to the left until stable
        x_position = 4
        while not check_stability():
            ab_drop('b13', x_position)
            x_position -= 1
        
        # Move 'b31' to the right until stable
        x_position = 7
        while not check_stability():
            ab_drop('b31', x_position)
            x_position += 1

# Call function to generate the required structure
place_bricks()
