# Map initialization
MAP = [[0]*20 for _ in range(16)]

# Helper function to check if a cell is occupied
def is_occupied(x, y):
    return MAP[y][x] == 1

# Helper function to check the stability of a brick
def is_stable(x, y, width, height):
    for i in range(width):
        left_count = 0
        right_count = 0
        for j in range(height):
            if is_occupied(x + i, y + j - 1):
                left_count += 1
            if is_occupied(x + i + 1, y + j - 1):
                right_count += 1
        if left_count != right_count:
            return False
    return True

# Function to drop a brick and update the map

    x_pos = y - width//2
    y_pos = 15 - height
    
    # Check for boundary collision
    if x_pos < 0 or x_pos + width > 20 or y_pos < 0:
        return

    # Check if any cell is occupied in the current position
    for i in range(width):
        for j in range(height):
            if is_occupied(x_pos + i, y_pos + j):
                return

    # Check stability before placing the brick
    if not is_stable(x_pos, y_pos, width, height):
        return
    
    # Place the brick on the map
    for i in range(width):
