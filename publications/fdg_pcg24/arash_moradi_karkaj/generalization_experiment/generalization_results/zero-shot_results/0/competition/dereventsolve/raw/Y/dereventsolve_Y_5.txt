def ab_drop(row):
    return 'ab.drop(' + str(row) + ')'

def generate_letter(letter):
    if letter not in database:
        return 'Letter not found in the database.'

    letter_structure = database[letter]
    ab_drop_calls = []
    for row in letter_structure:
        for i in range(len(row)):
            if row[i] == '*':
                ab_drop_calls.append(ab_drop(i))

    return '\n'.join(ab_drop_calls)
